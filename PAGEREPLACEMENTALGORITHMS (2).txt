import java.io.*;
import java.util.*;

class FIFO{
    int fc,pc;
    int pageFaults = 0;
    int pageHits = 0;
    int[] referencestring;
    FIFO(int pageCount,int framesCount,int[] pages){
        fc = framesCount;
        pc = pageCount;
        referencestring = pages;
    }
    void executeFIFO(){
        Queue<Integer> frames = new LinkedList<>();      //to store currently loaded pages
        System.out.println("\nFIFO Page Replacement Algorithm\n");
        for(int i = 0; i < pc ; i++){
            int currentpage = referencestring[i];
            if(frames.contains(currentpage)){
                pageHits++;
                System.out.println("Page " + currentpage + "--> HIT | Frames: " + frames);
            }else{
                pageFaults++;
                if(frames.size() == fc){
                    int removed = frames.poll(); //remove the oldest page(at index 0 - first page in queue)
                    System.out.println("FULL! Page " + removed + " removed from frames.");
                }
                frames.add(currentpage);
                System.out.println("Page " + currentpage + "--> FAULT | Frames: " + frames);
            }
        }
        System.out.println("\nTotal Page Faults: " + pageFaults);
        System.out.println("Total Page Hits: " + pageHits);
    }
}

class LRU{
    int fc,pc;
    int pageFaults = 0;
    int pageHits = 0;
    int[] referencestring;
    LRU(int pageCount,int framesCount,int[] pages){
        fc = framesCount;
        pc = pageCount;
        referencestring = pages;
    }
    void executeLRU(){
        List<Integer> frames = new ArrayList<>(fc);     //to store currently loaded pages
        System.out.println("\nLRU Page Replacement Algorithm\n");
        for(int i = 0; i < pc; i++){
            int currentpage = referencestring[i];
            if(frames.contains(currentpage)){
                pageHits++;
                //Move page to end(most recently used)(last page is most recently used)
                frames.remove(Integer.valueOf(currentpage));
                frames.add(currentpage);
                System.out.println("Page " + currentpage + "--> HIT | Frames: " + frames);
            }else{
                pageFaults++;
                if(frames.size() == fc){
                    int removed = frames.remove(0); //remove the least recently used page(at index 0 - first page in array)
                    System.out.println("FULL! Page " + removed + " removed from frames.");
                }
                frames.add(currentpage);
                System.out.println("Page " + currentpage + "--> FAULT | Frames: " + frames);
            }
        }
        System.out.println("\nTotal Page Faults: " + pageFaults);
        System.out.println("Total Page Hits: " + pageHits);
    }
}

class Optimal{
    int fc,pc;
    int pageFaults = 0;
    int pageHits = 0;
    int[] referencestring;
    Optimal(int pageCount,int framesCount,int[] pages){
        fc = framesCount;
        pc = pageCount;
        referencestring = pages;
    }
    void executeOptimal(){
        List<Integer> frames = new ArrayList<>(pc);
        System.out.println("\nOptimal Page Replacement Algorithm\n");
        for(int i = 0; i<pc;i++){
            int currentpage = referencestring[i];
            if(frames.contains(currentpage)){
                pageHits++;
                System.out.println("Page " + currentpage +"--> HIT | Frames: " + frames);
                continue;
            }
            pageFaults++;
            if(frames.size() < fc){
                frames.add(currentpage);

            }else{
                int farthestindex = -1;
                int pagetoreplace = -1;
                for(int page: frames){
                    int nextuse = Integer.MAX_VALUE;
                    for ( int j = i + 1; j < pc; j++){
                        if(referencestring[j] == page){
                            nextuse = j;
                            break;
                        }
                    }
                    if(nextuse == Integer.MAX_VALUE){
                        pagetoreplace = page;
                        break;
                    }
                    if(nextuse > farthestindex){
                        farthestindex = nextuse;
                        pagetoreplace = page;
                    }
                }
                frames.remove(Integer.valueOf(pagetoreplace));
                frames.add(currentpage);
            }
             System.out.println("Step " + (i + 1) + "Page " + currentpage +"--> FAULT | Frames: " + frames);
        }
        System.out.println("\nTotal Page Faults: " + pageFaults);
        System.out.println("Total Page Hits: " + pageHits);
    }
}
